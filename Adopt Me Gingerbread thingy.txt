local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/AikaV3rm/UiLib/master/Lib.lua')))() --[[ !!DONT REMOVE THIS!! this is pretty important and its basically the brains for the entire thing ]]--

local w = library:CreateWindow("Adopt Me")

local b = w:CreateFolder("Gingerbread Farm") -- Creates the folder(U will put here your buttons,etc)

local c = w:CreateFolder("Credits")  -- obviously u can make more than 1 folder within a window (this is just an example)

c:Label("MinaRizik or LunarNight#6165",{
   TextSize = 14; -- Self Explaining
   TextColor = Color3.fromRGB(255,255,255); -- Self Explaining
   BgColor = Color3.fromRGB(30, 30, 30); -- Self Explaining
})

b:Button("Start Autofarm",function()
game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(146.25727844238, 18.748636245728, -1525.3822021484)
wait(1)
_G.CollectGingerbreads = true
local Map = workspace.Interiors.MainMap
while wait(.001) do
  if _G.CollectGingerbreads then
      for i,v in pairs(Map:GetChildren()) do
          if v.Name:match('GingerbreadRig') then
              game:GetService('Players').LocalPlayer.Character.HumanoidRootPart.CFrame = v:FindFirstChildOfClass('MeshPart').CFrame
          end
      end
  end
end
end)

b:Button("Stop Autofarm",function()
 _G.CollectGingerbreads = false
local Map = workspace.Interiors.MainMap
while wait(.5) do
  if _G.CollectGingerbreads then
      for i,v in pairs(Map:GetChildren()) do
          if v.Name:match('GingerbreadRig') then
              game:GetService('Players').LocalPlayer.Character.HumanoidRootPart.CFrame = v:FindFirstChildOfClass('MeshPart').CFrame
          end
      end
  end
end
end)

b:Button("     Claim Gingerbread",function()
